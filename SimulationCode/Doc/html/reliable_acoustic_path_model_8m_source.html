34 function [loss, propagationTime] = transmissionLoss(obj,sendLocation, receiveLocation, f)
35 %compute distance in meters
36 d = norm(sendLocation - receiveLocation);
37 %never let two items be less than a meter apart
38 d = max(1,d);
39 %transmission loss 1/r^2
40 TL = 20 * log10(d);
41 %compute absorption loss
42 absorption = obj.absorption(f);
43 TL = TL + absorption * d;
44 %make a random failure happen
45 thisRealization = rand(1,1);
46 if thisRealization < obj.pFail
47 TL = TL + 200; %so much loss we'll never recover
48 end
49 loss = TL;
50 %and prop time
51 c=1500; %approximate speed of sound in seawater
52 propagationTime = d/c;
53 end
