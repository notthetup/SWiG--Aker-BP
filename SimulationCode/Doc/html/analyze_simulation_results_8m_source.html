5%analyze results from a simulation run

8%determine if any HD usage

14end

16%handle SMAC channel first
17SMACsentPacketInfo = sentPacketInfo(sentPacketInfo(:,1)<1e5,:);
18%first, find number of messages sent
19results.SMACnumMessagesSent = size(SMACsentPacketInfo,1);
20%find number of lost messages and accumulate latency array
21latencies = nan(size(SMACsentPacketInfo,1),1);

24results.SMACnumAckRequiredMessagesLost = 0;

32 if ~isempty(which)
33 latencies(i) = receivedPacketInfo(which,3) - SMACsentPacketInfo(i,3);
34 SMACtotalBits = SMACtotalBits + SMACsentPacketInfo(i,5);
35 if HD
36 if SMACsentPacketInfo(i,6)>0.95
37 SMACOutsideHDBits = SMACOutsideHDBits + SMACsentPacketInfo(i,5);
38 else
39 SMACduringHDBits = SMACduringHDBits + SMACsentPacketInfo(i,5);
40 end
41 end
42 else
43 results.SMACnumMessagesLost = results.SMACnumMessagesLost + 1;
44 if (sentPacketInfo(i,2))

46 end
47 end
48end
49ack = latencies(isfinite(latencies) & ackRequired);
50nack = latencies(isfinite(latencies)& ~ackRequired);
51results.SMACmeanLatency = [mean(nack) mean(ack)];
52results.SMACsigmaLatency = [std(nack) std(ack)];
53results.SMACmedianLatency = [median(nack) median(ack)];

55results.SMACminLatency = [min(nack) min(ack)];
56results.SMACrawLatency = sort(latencies);

59 %now handle HD channel
60 HDsentPacketInfo = sentPacketInfo(sentPacketInfo(:,1)>=1e5,:);

69 results.HDnumMessagesLost = results.HDnumMessagesLost + 1;
70 else
71 HDtotalBits = HDtotalBits + HDsentPacketInfo(i,5);
72 end
73 end
74 results.SMACNoHDThroughput = SMACOutsideHDBits/(range(SMACsentPacketInfo(:,3)) - (endHD - startHD));
75 results.SMACDuringHDThroughput = SMACduringHDBits / (endHD - startHD);
