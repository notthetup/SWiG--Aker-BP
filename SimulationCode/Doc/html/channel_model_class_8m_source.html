46 function result = absorption(obj,fHz)
47 %set up simple variables to make entering formulas easier
48 T = obj.temperature;
49 S = obj.salinity;
50 f = fHz / 1e3; %needed in kHz for the formulas
51 z = obj.depth;
52 PH = obj.pH;
53 %calculate effective speed of sound
54 c = 1412 + 3.21*T + 1.19*S + 0.0167*z;
55 %boric acid paramters
56 A1 = (8.86/c) * (10^(0.78*PH - 5));
57 %pH modifier
58 P1 = 1;
59 f1 = 2.8*sqrt(S/35) * (10^(4 - (1245/(T+273))));
60 %magnesium sulphate parameters
61 A2 = 21.44*(S/c) * (1 + 0.025*T);
62 P2 = 1 - 1.37e-4 *z + 6.2e-9*z*z;
63 f2 = (8.17 * (10^(8-(1990/(T+273)))))/(1+0.0018*(S-35));
64 %water viscosity parameters
65 P3 = 1 -3.83e-5*z + 4.9e-10*z*z;
66 if T < 20
67 A3 = 4.937e-4 - 2.59e-5*T + 9.11e-7*T*T - 1.5e-8*T*T*T;
68 else
69 A3 = 3.964e-4 - 1.146e-5*T + 1.45e-7*T*T - 6.5e-10*T*T*T;
70 end
71 %put them all together
72 alpha = A1*P1*(f1*f*f/(f1*f1+f*f)) + ...
73 A2*P2*(f2*f*f/(f2*f2+f*f)) + ...
74 A3*P3*f*f;
75 %and convert back
76 result = alpha * 1e-3;
77 end
